每个应用最低支持的系统版本

1

find /Applications -maxdepth 2 -name "*.app" -exec sh -c 'for app; do name=$(basename "$app" .app); ver=$(/usr/libexec/PlistBuddy -c "Print :LSMinimumSystemVersion" "$app/Contents/Info.plist" 2>/dev/null || echo N/A); echo "$name: $ver"; done' _ {} + | sort
