TS 装饰器的使用

Feb 24, 2025
阅读时长：7 分钟

最近在看一个 SaaS 项目，项目中的数据请求函数有点另类，不同于我之前写的项目；它的每个请求函数是基于 装饰器 来构建的。相比于继承，装饰器的控制颗粒度更小更灵活，并且对类本身的侵入性较小。

本文中，代码的运行时为 TypeScript@5.8.2 。

装饰器

TS 装饰器分为：

  1. 类装饰器；
  2. 属性装饰器；
  3. 方法装饰器；
  4. 参数装饰器；
  5. 访问器装饰器。

@ClassDecoratorFn
class DecoratorsExample {
  sayHi!: () => void;
}
/**
 * 类装饰器的类型：
 * node_modules/typescript/lib/lib.decorators.legacy.d.ts
 * declare type ClassDecorator = <TFunction extends Function>(target: TFunction) => TFunction | void;
 */
function ClassDecoratorFn(target: Function): void {
  /**
   * [Function: DecoratorsExample]
   * target 指向「构造函数」
   */
  console.log('ClassDecoratorFn > target', target === DecoratorsExample); // true
  // 可以添加｜修改｜删除原型上的属性和方法
  target.prototype.name = 'DecoratorsExample';
  target.prototype.sayHi = function () {
    // 借助原型上的方法，输出日志
    this.output(`I'M ${this.name}`);
  };
  target.prototype.output = (msg: string) => {
    console.log(msg);
  };
}
/**
 * 类装饰器相当于在这里调用了 ClassDecoratorFn(DecoratorsExample)
 * 这也说明，装饰器是可以叠加的，它只是对被修饰对象进行了修改
 */
const example = new DecoratorsExample();
example.sayHi(); // I'M DecoratorsExample

class DecoratorsExample {
  @PropertyDecoratorFn
  sex: 'male' | 'female' = 'male';
}
/**
 * 属性装饰器的类型：
 * node_modules/typescript/lib/lib.decorators.legacy.d.ts
 * declare type PropertyDecorator = (target: Object, propertyKey: string | symbol) => void;
 */
function PropertyDecoratorFn(target: Object, propertyKey: string | symbol) {
  console.log('PropertyDecoratorFn > target', target); // 指向实例 DecoratorsExample: {}
  console.log('PropertyDecoratorFn > propertyKey', propertyKey); // 指向属性名称 sex
  Object.defineProperty(target, propertyKey, {
    configurable: true,
    enumerable: true,
    get: () => {
      return 'female';
    },
    set: value => {}
  });
}
const example = new DecoratorsExample();
console.log(example.sex); // 属性值被 PropertyDecoratorFn 改写为 female

class DecoratorsExample {
  private sex: 'male' | 'female' = 'male';

  @InstanceMethodDecoratorFn
  public getSex() {
    return this.sex;
  }

  @StaticMethodDecoratorFn
  static getInfo() {
    return null;
  }
}
/**
 * 方法装饰器的类型：
 * node_modules/typescript/lib/lib.decorators.legacy.d.ts
 * declare type MethodDecorator = <T>(target: Object, propertyKey: string | symbol, descriptor: TypedPropertyDescriptor<T>) => TypedPropertyDescriptor<T> | void;
 */
function InstanceMethodDecoratorFn(
  target: Object,
  propertyKey: string | symbol,
  descriptor: PropertyDescriptor
): PropertyDescriptor | void {
  console.log('InstanceMethodDecoratorFn > target', target); // 指向实例 DecoratorsExample: {}
  console.log('InstanceMethodDecoratorFn > propertyKey', propertyKey); // 指向方法名称 getSex
  /**
   * 返回对应的「元数据」信息
   * {
   *   "configurable": true,
   *   "enumerable": false,
   *   "writable": true
   * }
   */
  console.log('InstanceMethodDecoratorFn > descriptor', descriptor); // 返回「数据属性」
  // 修改「数据属性」
  return {
    configurable: false,
    enumerable: false,
    writable: false
  };
}
function StaticMethodDecoratorFn(
  target: Object,
  propertyKey: string | symbol,
  descriptor: PropertyDescriptor
): PropertyDescriptor | void {
  console.log('StaticMethodDecoratorFn > target', target); // 指向「构造函数」
  console.log('StaticMethodDecoratorFn > propertyKey', propertyKey); // 指向方法名称 getInfo
  /**
   * 返回对应的「元数据」信息
   * {
   *   "configurable": true,
   *   "enumerable": false,
   *   "writable": true
   * }
   */
  console.log('StaticMethodDecoratorFn > descriptor', descriptor);
}
const example = new DecoratorsExample();
// 会在运行时报错，因为修改「数据属性」之后，getSex 不能被修改
example.getSex = function () {
  return 'male';
};

function SayDecorator<T>(
  target: T,
  propertyKey: keyof T,
  descriptor: PropertyDescriptor
) {
  const originalMethod = target[propertyKey];
  // 重写原方法
  descriptor.value = function () {
    if (typeof originalMethod === 'function') {
      console.log(`${originalMethod.apply(target)}, I'M 12 Years Old`);
    }
  };
}

example.hi(); // Hi, I'M DecoratorsExample, I'M 12 Years Old

装饰器工厂

一般很少直接使用一个装饰器函数，因为它的使用场景太固定了；在实际业务中，通常使用工厂函数来返回装饰器（有点函数柯理化的味道）。这样装饰器能够通过闭包，访问到更多的配置信息；同时也方便将属于相同业务的装饰器集中起来：

enum DecoratorFactoryType {
  ClassDecorator = 'ClassDecorator',
  PropertyDecorator = 'PropertyDecorator',
  MethodDecorator = 'MethodDecorator',
  ParameterDecorator = 'ParameterDecorator'
}
function DecoratorFactory(type: DecoratorFactoryType) {
  switch (type) {
    case DecoratorFactoryType.ClassDecorator:
      return function ClassDecoratorFn(target: Function): Function | void {};
    case DecoratorFactoryType.PropertyDecorator:
      return function PropertyDecoratorFn(
        target: Object,
        propertyKey: string | symbol
      ): void {};
    case DecoratorFactoryType.MethodDecorator:
      return function MethodDecoratorFn(
        target: Object,
        propertyKey: string | symbol,
        descriptor: PropertyDescriptor
      ): PropertyDescriptor | void {};
    default:
      return function ParameterDecoratorFn(
        target: Object,
        propertyKey: string | symbol | undefined,
        parameterIndex: number
      ): void {};
  }
}

function DelayInvoke<T>(time: number = 1000) {
  return (target: T, propertyKey: keyof T, descriptor: PropertyDescriptor) => {
    console.log('sayHi Has Been Modified');
    const originalMethod = target[propertyKey];
    descriptor.value = (...args: unknown[]) => {
      setTimeout(() => {
        if (typeof originalMethod === 'function') {
          return originalMethod.apply(target, args);
        }
      }, time);
    };
  };
}
class DelayInvokeExample {
  @DelayInvoke<DelayInvokeExample>(3000)
  sayHi() {
    console.log("Hi, I'M Delay Invoked");
  }
}
const example = new DelayInvokeExample();
example.sayHi(); // 3 秒后打印 Hi, I'M Delay Invoked

class AuthCatchExample {
  currentAccountPermission = 'User';

  @(CatchError(AuthCatchExample)<AuthCatchExample>)
  @(AdminPermission('Admin')<AuthCatchExample>)
  aRequireAdminFn() {
    console.log('I Need A Admin Permission To Run');
  }
}
function AdminPermission(permitPermission: string) {
  return function <T extends Object>(
    target: T,
    propertyKey: keyof T,
    descriptor: PropertyDescriptor
  ) {
    const { currentAccountPermission = 'User' } = target as any;
    const originalMethod = target[propertyKey];
    descriptor.value = (...args: unknown[]) => {
      if (currentAccountPermission !== permitPermission) {
        throw new Error('Insufficient Permission');
      } else if (typeof originalMethod === 'function') {
        return originalMethod.apply(target, args);
      }
    };
  };
}
function CatchError(from: Function) {
  return function <T extends Object>(
    target: T,
    propertyKey: keyof T,
    descriptor: PropertyDescriptor
  ) {
    const { name } = from;
    const originalMethod = target[propertyKey];
    descriptor.value = (...args: unknown[]) => {
      try {
        if (typeof originalMethod === 'function') {
          return originalMethod.apply(target, args);
        }
      } catch (error) {
        // 可以将错误上报
        console.log(`Error From ${name}: `, error);
      }
    };
  };
}
const example = new AuthCatchExample();
example.aRequireAdminFn();

如果运行这段代码，你会发现 AdminPermission 装饰器并没有起作用；原因是，每次我们获取原始方法时，都是通过 target[propertyKey] 拿的。

class RequestBuilder {
  // 用来设置请求体
  static body() {
    return (
      target: Record<string, any>,
      propertyKey: string | symbol | undefined,
      parameterIndex: number
    ): void => {
      const privateKey = `private_body_${propertyKey?.toString()}`;
      const privateMeta = {
        key: 'body',
        parameterIndex
      };
      if (Array.isArray(target[privateKey])) {
        target[privateKey].push(privateMeta);
      } else {
        target[privateKey] = [privateMeta];
      }
    };
  }
  // 用来设置请求头
  static headers() {
    return (
      target: Record<string, any>,
      propertyKey: string | symbol | undefined,
      parameterIndex: number
    ): void => {
      const privateKey = `private_headers_${propertyKey?.toString()}`;
      const privateMeta = {
        key: 'headers',
        parameterIndex
      };
      if (Array.isArray(target[privateKey])) {
        target[privateKey].push(privateMeta);
      } else {
        target[privateKey] = [privateMeta];
      }
    };
  }
  // 用来设置路径参数
  static path(key: string) {
    return (
      target: Record<string, any>,
      propertyKey: string | symbol | undefined,
      parameterIndex: number
    ): void => {
      const privateKey = `private_path_${propertyKey?.toString()}`;
      const privateMeta = {
        key,
        parameterIndex
      };
      if (Array.isArray(target[privateKey])) {
        target[privateKey].push(privateMeta);
      } else {
        target[privateKey] = [privateMeta];
      }
    };
  }
  // 用来设置查询参数

    return (
      target: Record<string, any>,
      propertyKey: string | symbol | undefined,
      parameterIndex: number
    ): void => {
      const privateKey = `private_query_${propertyKey?.toString()}`;
      const privateMeta = {
        key,
        parameterIndex
      };
      if (Array.isArray(target[privateKey])) {
        target[privateKey].push(privateMeta);
      } else {
        target[privateKey] = [privateMeta];
      }
    };
  }
}

class RequestBuilder {
  static body(): ParameterDecorator {
    return RequestBuilder.GeneratePayload('body');
  }
  static headers(): ParameterDecorator {
    return RequestBuilder.GeneratePayload('headers');
  }
  static path(key: string): ParameterDecorator {
    return RequestBuilder.GeneratePayload('path', key);
  }
  static query(key: string): ParameterDecorator {
    return RequestBuilder.GeneratePayload('query', key);
  }
  static GeneratePayload(
    where: 'body' | 'headers' | 'path' | 'query',
    key?: string
  ) {
    // 返回参数装饰器
    return (
      target: Record<string, any>,
      propertyKey: string | symbol | undefined,
      parameterIndex: number
    ): void => {
      const privateKey = `private_${where}_${propertyKey?.toString()}`;
      const privateMeta = key
        ? {
            key,
            parameterIndex
          }
        : { parameterIndex };
      if (where === 'body' || where === 'headers') {
        target[privateKey] = privateMeta;
      } else {
        if (Array.isArray(target[privateKey])) {
          target[privateKey].push(privateMeta);
        } else {
          target[privateKey] = [privateMeta];
        }
      }
    };
  }
}

class RequestBuilder {
  static getMethod(roughUrl: string, options: RequestInit): MethodDecorator {
    return RequestBuilder.GenerateRequest(roughUrl, 'GET', options);
  }
  static deleteMethod(roughUrl: string, options: RequestInit): MethodDecorator {
    return RequestBuilder.GenerateRequest(roughUrl, 'DELETE', options);
  }
  static postMethod(roughUrl: string, options: RequestInit): MethodDecorator {
    return RequestBuilder.GenerateRequest(roughUrl, 'POST', options);
  }
  static putMethod(roughUrl: string, options: RequestInit): MethodDecorator {
    return RequestBuilder.GenerateRequest(roughUrl, 'PUT', options);
  }
  static GenerateRequest(
    roughUrl: string,
    method: 'GET' | 'POST' | 'DELETE' | 'PUT',
    options: RequestInit
  ) {
    return (
      target: Record<string, any>,
      propertyKey: string | symbol | undefined,
      descriptor: PropertyDescriptor
    ): PropertyDescriptor | void => {
      const { parameterIndex: bodyParameterIndex } =
        target[`private_body_${propertyKey?.toString()}`] || {};
      const { parameterIndex: headersParameterIndex } =
        target[`private_headers_${propertyKey?.toString()}`] || {};
      const pathArray: Array<Record<string, any>> =
        target[`private_path_${propertyKey?.toString()}`] || [];
      const queryArray: Array<Record<string, any>> =
        target[`private_query_${propertyKey?.toString()}`] || [];
      descriptor.value = function (...args: Array<any>): Promise<Response> {
        let body = args[bodyParameterIndex];
        let headers = args[headersParameterIndex];
        let url = roughUrl;
        pathArray.forEach(({ key, parameterIndex }) => {
          url = url.replace(`{${key}}`, args[parameterIndex]?.toString());
        });
        let query = '';
        queryArray.forEach(({ key, parameterIndex }) => {
          query = `${query}${key}=${args[parameterIndex]}&`;
        });
        query = query.slice(0, -1);
        url = `${url}?${query}`;
        return fetch(url, {
          ...options,
          body: method === 'GET' ? null : JSON.stringify(body),
          headers,
          method
        } satisfies RequestInit);
      };
    };
  }
}
