Function.prototype.apply = function (context = window, args = []) {
  if (typeof this !== 'function') {
    throw new Error('apply 方法只能在函数上调用');
  }
  const fn = Symbol('fn');
  context[fn] = this;
  const result = context[fn](...args);
  delete context[fn];
  return result;
};
Function.prototype.call = function (context = window, ...args) {
  if (typeof this !== 'function') {
    throw new Error('call 方法只能在函数上调用');
  }
  const fn = Symbol('fn'); // 声明一个不等于任何值的变量
  context[fn] = this; // 将 example 作为待绑定对象的一个方法
  const result = context[fn](...args); // 保存调用结果
  delete context[fn]; // 删除手动添加的属性
  return result; // 将结果返回
};
Function.prototype.bind = function (context = window, ...params1) {
  if (typeof this !== 'function') {
    throw new Error('bind 方法只能在函数上调用');
  }
  const that = this;
  function bindFn(...params2) {
    // 判断是否用于构造函数
    if (this instanceof bindFn) {
      // new 的优先级比 bind 高，new 出来的对象的构造函数仍然指向 that
      return new that(...params1, ...params2);
    }
    return that.call(context, ...params1, ...params2);
  }
  return bindFn;
};

function compose(...fns) {
  const that = this;
  let { length } = fns;
  let result = null;
  return function temp(params) {
    if (length > 0) {
      result = fns[--length].call(that, params);
      return temp.call(that, result);
    } else {
      return result;
    }
  };
}

function instanceOf(L, R) {
  if (L === null || L === undefined) {
    return false;
  }
  L = Object.getPrototypeOf(L);
  R = R.prototype;
  while (L) {
    if (L === R) {
      return true;
    }
    L = Object.getPrototypeOf(L);
  }
  return false;
}

console.log(instanceOf({}, Object)); // true
console.log(instanceOf([], Array)); // true

Array.prototype.implementMapUsingReduce = function (fn) {
  const list = this;
  if (!Array.isArray(list)) {
    throw new Error('不能在非数组上调用 map');

  const list = this;
  if (!Array.isArray(list)) {

  let value = initValue === undefined ? list[0] : initValue;

function setInterval(cb, time) {
  let getNow = () => Date.now();
  let startTime = getNow();
  let endTime = startTime;
  let timer;
  let loop = () => {
    timer = window.requestAnimationFrame(loop);
    endTime = getNow();
    if (endTime - startTime > time) {
      endTime = startTime = getNow();
      cb(timer);
      // window.cancelAnimationFrame(timer) // 如果是 setTimeout，放开这一行
    }
  };
  timer = window.requestAnimationFrame(loop);
  return timer;
}

class EventEmitter {
  _events = Object.create(null);
  add(type, handler) {
    if (!this._events[type]) {
      this._events[type] = [];
    }
    this._events[type].push(handler);
  }
  remove(type, handler) {
    if (!this._events[type]) {
      return;
    }
    if (handler) {
      this._events[type] = this._events[type].filter(item => item !== handler);
    } else {
      delete this._events[type];
    }
  }
  once(type, handler) {
    const temp = (...args) => {
      handler(...args);
      this.remove(type, temp);
    };
    this.add(type, temp);
  }
  emit(type, ...args) {
    if (!this._events[type]) {
      return;
    }
    this._events[type].forEach(handler => {
      handler(...args);
    });
  }
}

class LazyManClass {
  constructor(name) {
    this.name = name;
    this.doSomething = [];
    console.log(`Hi I am ${this.name}`);
    setTimeout(() => {
      this.next();
    }, 0);
  }
  eat(food) {
    const sth = (food => () => {
      console.log(`I am eating ${food}`);
      this.next();
    })(food);
    this.doSomething.push(sth);
    return this;
  }
  sleep(time) {
    const sth = (time => () => {
      setTimeout(() => {
        console.log(`等待了${time}秒...`);
        this.next();
      }, time * 1000);
    })(time);
    this.doSomething.push(sth);
    return this;
  }
  sleepFirst(time) {
    var sth = (time => () => {
      setTimeout(() => {
        console.log(`等待了${time}秒...`);
        this.next();
      }, time * 1000);
    })(time);
    // 将任务添加到最前边
    this.doSomething.unshift(sth);
    return this;
  }
  next() {
    let doSth = this.doSomething.shift();
    typeof doSth === 'function' && doSth();
  }
}
function LazyMan(name) {
  return new LazyManClass(name);
}

function MyPromise(executor) {
  const self = this;
  self.data = undefined;
  self.status = 'pending';
  self.onFulfilledCbs = [];
  self.onRejectedCbs = [];
  function fulfill(value) {
    if (self.status !== 'pending') {
      return;
    }
    self.status = 'fulfilled';
    self.data = value;
    self.onFulfilledCbs.forEach(fn => fn(value));
  }
  function reject(reason) {
    if (self.status !== 'pending') {
      return;
    }
    self.status = 'rejected';
    self.data = reason;
    self.onRejectedCbs.forEach(fn => fn(reason));
  }
  try {
    executor(fulfill, reject);
  } catch (e) {
    reject(e);
  }
}
MyPromise.prototype.then = function (onFulFilled, onRejected) {
  const self = this;
  let promise = undefined;
  onFulFilled =
    typeof onFulFilled === 'function'
      ? onFulFilled
      : function (v) {
          return v;
        };
  onRejected =
    typeof onRejected === 'function'
      ? onRejected
      : function (e) {
          throw e;
        };
  if (self.status === 'pending') {
    promise = new MyPromise(function (fulfill, reject) {
      self.onFulfilledCbs.push(function () {
        try {
          const temp = onFulFilled(self.data);
          if (temp instanceof MyPromise) {
            temp.then(fulfill, reject);
          }
          fulfill(temp);
        } catch (e) {
          reject(e);
        }
      });
      self.onRejectedCbs.push(function () {
        try {
          const temp = onRejected(self.data);
          if (temp instanceof MyPromise) {
            temp.then(fulfill, reject);
          }
        } catch (e) {
          reject(e);
        }
      });
    });
  }
  if (self.status === 'fulfilled') {
    promise = new MyPromise(function (fulfill, reject) {
      try {
        const temp = onFulfilled(self.data);
        if (temp instanceof MyPromise) {
          temp.then(fulfill, reject);
        }
        fulfill(temp);
      } catch (e) {
        reject(e);
      }
    });
  }
  if (self.status === 'rejected') {
    promise = new MyPromise(function (fulfill, reject) {
      try {
        const temp = onRejected(self.data);
        if (temp instanceof MyPromise) {
          temp.then(fulfill, reject);
        }
      } catch (e) {
        reject(e);
      }
    });
  }
  return promise;
};
MyPromise.prototype.catch = function (onRejected) {
  return this.then(null, onRejected);
};

Promise.myAll = ps => {
  if (typeof ps[Symbol.iterator] !== 'function') {
    return Promise.reject('arguments must be an iterator object');
  }
  let resolvedList = Array(ps.length);
  let resolvedNum = 0;
  return new Promise((resolve, reject) => {
    ps.forEach((p, i) => {
      Promise.resolve(p)
        .then(v => {
          resolvedList[i] = v;
          resolvedNum = resolvedNum + 1;
          if (resolvedList.length === resolvedNum) {
            resolve(resolvedList);
          }
        })
        .catch(reject);
    });
  });
};
Promise.myRace = ps => {
  if (typeof ps[Symbol.iterator] !== 'function') {
    return Promise.reject('arguments must be an iterator object');
  }
  return new Promise((resolve, reject) => {
    ps.forEach(p => {
      Promise.resolve(p).then(resolve).catch(reject);
    });
  });
};
Promise.myRetry = (p, time = 3) => {
  return new Promise(async (resolve, reject) => {
    while (time--) {
      try {
        const v = await p();
        resolve(v);
        break;
      } catch (e) {
        if (!time) {
          reject(e);
        }
      }
    }
  });
};
Promise.myAllSettled = ps => {
  if (typeof ps[Symbol.iterator] !== 'function') {
    return Promise.reject('arguments must be an iterator object');
  }
  let tempList = Array(ps.length);
  let tempNum = 0;
  return new Promise((resolve, reject) => {

          tempList[i] = { status: 'fulfilled', value: v };

        })
        .finally(() => {
          tempNum = tempNum + 1;
          if (tempList.length === tempNum) {
            resolve(tempList);
          }
        });
    });
  });
};

function deepClone(o, map = new WeakMap()) {
  const type = o =>
    Object.prototype.toString
      .call(o)
      .replace(/\[object |\]/g, '')
      .toLowerCase();
  if (
    type(o) === 'null' ||
    type(o) === 'undefined' ||
    type(o) === 'string' ||
    type(o) === 'number' ||
    type(o) === 'boolean' ||
    type(o) === 'symbol' ||
    type(o) === 'function'
  ) {
    return o;
  }
  // https://developer.chrome.com/docs/apps/contentSecurityPolicy/#relaxing-inline-script
  // if (type(o) === 'function') {
  //   return eval(o.toString());
  // }
  if (type(o) === 'date') {
    return new Date(o);
  }
  if (type(o) === 'regexp') {
    return new RegExp(o);
  }
  if (map.has(o)) {
    return map.get(o);
  }
  let temp = new o.constructor();
  map.set(o, temp);
  // WeakMap 不能被遍历，所以不能被拷贝
  if (type(o) === 'map') {
    o.forEach((v, k) => {
      temp.set(k, deepClone(v, map));
    });
    return temp;
  }
  // WeakSet 不能被遍历，所以不能被拷贝
  if (type(o) === 'set') {
    o.forEach(v => {
      temp.add(deepClone(v, map));
    });
    return temp;
  }
  // Object、Array
  for (let i in o) {
    if (o.hasOwnProperty(i)) {
      temp[i] = deepClone(o[i], map);
    }
  }
  return temp;
}

function throttle(fn, time = 300) {
  let previous = 0;
  return function (...args) {
    const context = this;
    let now = Date.now();
    if (now - previous > time) {
      previous = now;
      fn.apply(context, args);
    }
  };
}

function debounce(fn, wait = 300, isImmediate) {
  let timer = null;
  return function (...args) {
    const context = this;
    if (timer) {
      clearTimeout(timer);
    }
    if (isImmediate && !timer) {
      fn.apply(context, args);
    }
    timer = setTimeout(() => {
      fn.apply(context, args);
    }, wait);
  };
}

function hasCycle(linkedList) {
  // 分别取两个指针（快、慢），如果成环，那么快、慢指针一定会相遇
  if (!linkedList.head) return false;
  let fast = linkedList.head;
  let slow = linkedList.head;
  while (slow.next && fast.next.next) {
    slow = slow.next;
    fast = fast.next.next;
    if (fast === slow) {
      break;
    }
  }
  if (!fast) {
    return false;
  }
  slow = linkedList.head;
  while (slow !== fast) {
    // 重新相遇的点就是入环节点
    slow = slow.next;
    fast = fast.next;
  }
  return slow;
}

  // 从左到右一张一张处理这些扑克牌，最终要把这些牌分成若干堆
  for (let i = 0; i < array.length; i++) {
    const element = array[i]; // 要处理的扑克牌
    let right = result;
    let left = 0;
    // 左堆和右堆进行二分搜索，因为堆顶是有序排的，最终找到该牌要插入的堆
    while (left < right) {
      const mid = (left + right) >> 1;
      if (top[mid] > element) {
        right = mid;
      } else if (top[mid] < element) {
        left = mid + 1;
      } else {
        right = mid;
      }
    }
    /**
     * 只能把点数小的牌压到点数比它大的牌上
     * 如果当前牌点数较大没有可以放置的堆，则新建一个堆，把这张牌放进去
     * 如果当前牌有多个堆可供选择，则选择最左边的那一堆放置
     */
    if (left === result) {
      result++;
    }
    top[left] = element;
  }
  // 牌的堆数就是最长递增子序列的长度
  return result;
}

function compareVersion(v1, v2) {
  const v1Arr = `${v1}`
    .split('.')
    .map(item => (item === '0' ? '0' : item.replace(/^0+/gim, '')))
    .map(Number);
  const v2Arr = `${v2}`
    .split('.')
    .map(item => (item === '0' ? '0' : item.replace(/^0+/gim, '')))
    .map(Number);
  let i = 0;
  while (i < Math.min(v1Arr.length, v2Arr.length)) {
    const v1Item = v1Arr[i];
    const v2Item = v2Arr[i];
    if (v1Item > v2Item) {
      return 1;
    } else if (v1Item < v2Item) {
      return -1;
    }
    i++;
  }
  if (v1Arr.length) {
    return v1Arr.slice(i).every(item => item === 0) ? 0 : 1;
  }
  if (v2Arr.length) {
    return v2Arr.slice(i).every(item => item === 0) ? 0 : 1;
  }
  return 0;
}

class Trie {
  root = new TrieNode();
  // prefix 是否在前缀树中
  startsWith(prefix: string) {
    let node = this.root;
    for (let i = 0; i < prefix.length; i++) {
      const char = prefix[i];
      if (!node.children[char]) {
        return false;
      }
      node = node.children[char];
    }
    return true;
  }
  // 将 word 插入前缀树中
  insert(word: string) {
    let node = this.root;
    for (let i = 0; i < word.length; i++) {
      const char = word[i];
      if (!node.children[char]) {
        node.children[char] = new TrieNode();
      }
      node = node.children[char];
    }
    node.isEndOfWord = true;
  }
  // word 是否存在于前缀树中
  search(word: string) {
    let node = this.root;
    for (let i = 0; i < word.length; i++) {
      const char = word[i];
      if (!node.children[char]) {
        return false;
      }
      node = node.children[char];
    }
    return node.isEndOfWord;
  }
}

/**
 * 需要对原数组的某个区间的元素
 * 进行频繁的修改（增减）
 * 比如，给索引为 [2, 6] 的元素全部加 1
 * 再给索引为 [3, 9] 的元素全部减 3
 * 再给索引为 [0, 4] 的元素全部加 6
 * 最后，原数组是什么个情况
 */
class DifferenceArray {
  diffArr = [];
  originArr = [];
  constructor(originArr = []) {
    this.originArr = originArr;
    this.originArr.forEach((item, index, arr) => {
      if (index === 0) {
        this.diffArr[0] = item;
      } else {
        this.diffArr[index] = item - arr[index - 1];
      }
    });
  }
  increment(i, j, value) {
    this.diffArr[i] = this.diffArr[i] + value;
    if (j + 1 < this.diffArr.length) {
      this.diffArr[j + i] = this.diffArr[j + i] - value;
    }
  }
  result() {
    const temp = [];
    this.diffArr.forEach((item, index) => {
      if (index === 0) {
        temp[0] = item;
      } else {
        temp[index] = temp[index - 1] + item;
      }
    });
    return temp;
  }
}

function object2xml(object) {
  function loop(o, level = 1) {
    let xml = '';
    for (let k in o) {
      const tagName = k;
      const tagValue = o[k];
      const indent = '  '.repeat(level);
      if (typeof tagValue === 'object' && tagValue !== null) {
        // tagValue 为标签
        xml += `${indent}<${tagName}>\n`;
        xml += loop(tagValue, level + 1);
        xml += `${indent}</${tagName}>\n`;
      } else if (tagValue !== undefined) {
        // tagValue 为 TextNode
        xml += `${indent}<${tagName}>`;
        xml += tagValue;
        xml += `</${tagName}>\n`;
      }
    }
    return xml;
  }
  return `<root>\n${loop(object)}</root>`;
}

function mergeArr3(array) {
  /**
   * 每个元素都有自己所在的组织，组织中都有一个元素作为 Boss
   * 元素通过 findMyBoss 查找自己的 Boss
   * 如果元素的 Boss 是另一个组织的小弟
   * 那么元素的 Boss 为 Boss 的 Boss
   */
  const myBoss = new Map();
  function findMyBoss(item) {
    if (!myBoss.has(item)) {
      // 没找到 Boss，Boss 就是我自己
      myBoss.set(item, item);
    }
    if (myBoss.get(item) !== item) {
      /**
       * 如果 Boss 不是我自己，找找看 Boss 的 Boss
       * 最终的 Boss 为 Boss 的 Boss
       */
      myBoss.set(item, findMyBoss(myBoss.get(item)));
    }
    return myBoss.get(item); // 返回自己的 Boss
  }
  // 合并两个元素所在的组织
  function union(item1, item2) {
    const item1Boss = findMyBoss(item1); // item1 的 Boss
    const item2Boss = findMyBoss(item2); // item2 的 Boss
    // 说明两个元素还属于别的组织
    if (item1Boss !== item2Boss) {
      /**
       * Boss 不一样，两个组织应该合并
       * 让 item2Boss 成为新组织中的小弟
       * 让 item1Boss 做合并后的组织的 Boss
       */
      myBoss.set(item2Boss, item1Boss);
    }
  }
  // 遍历数组，每个元素找自己的终极 Boss
  for (let subArray of array) {
    if (!subArray?.length) continue;
    // 每个组织的第一个元素当 Boss
    const currentSubArrayBoss = subArray[0];
    for (let i = 1; i < subArray.length; i++) {
      union(currentSubArrayBoss, subArray[i]);
    }
  }
  // 每个元素都去找自己的终极 Boss 建立组织
  const group = new Map();
  for (let subArr of array) {
    for (let item of subArr) {
      const itemBoss = findMyBoss(item);
      if (!group.has(itemBoss)) {
        group.set(itemBoss, new Set());
      }
      // 将 Boss 和自己都加进去
      group.get(itemBoss).add(itemBoss).add(item);
    }
  }
  return [...group.values()].map(item => [...item]);
}

class Scheduler {
  limit = 2;
  queue = [];
  running = 0;
  add(promiseCreator) {
    const task = () => {
      this.running++;
      promiseCreator().finally(() => {
        this.running--;
        this.run();
      });
    };
    this.queue.push(task);
    if (this.running < this.limit) {
      this.run();
    }
  }
  run() {
    if (this.running < this.limit && this.queue.length) {
      const task = this.queue.shift();
      task?.();
    }
  }
}

  new Promise(resolve => {
    setTimeout(resolve, time);
  });

/**
 * 限流和防抖、节流的区别：
 * 限流，限制回调在一定时间内的调用次数
 * 防抖，在一定时间内，对调用进行去重
 * 节流，降低回调触发频率
 */
function rateLimitFn1(cb, limitCount = 1, limitTime = 1000) {
  /**
   * 存在问题：
   * 0.9 秒执行 limitCount 次，1.1 秒执行 limitCount 次
   * 最终，0.2 秒内执行了 limitCount*2 次
   */
  let currentCount = 0;
  let currentTime = Date.now();
  return function (...args) {
    const now = Date.now();
    if (now - currentTime > limitTime) {
      // currentTime 过期，重置状态
      currentCount = 0;
      currentTime = now;
    }
    if (currentCount < limitCount) {
      // 未达到阈值，触发回掉
      currentCount++;
      return cb.apply(this, args);
    }
    // 未过期，并达到阈值，什么也不做
  };
}

function rateLimitFn2(cb, limitCount = 1, limitTime = 1000) {
  const timeQueue = [];
  return function (...args) {
    const now = Date.now();
    while (timeQueue.length && now - timeQueue[0] > limitTime) {
      timeQueue.shift();
    }
    if (timeQueue.length < limitCount) {
      timeQueue.push(now);
      return cb.apply(this, args);
    }
    // 未过期，并达到阈值，什么也不做
  };
}

function findNoRepeatLength(str) {
  if (str.length === 0 || str.length === 1) {
    return str.length;
  }
  const temp = {}; // 记录每个字符最后出现的下标
  let result = 0;
  // 初始化两个指针，分别表示指向子字符串的首尾索引
  for (let i = 0, j = 0; j < str.length; j++) {
    /**
     * 如果子字符串的结尾的索引（之前）出现过
     * [i, j] 中存在了重复字符，i 移到区间外
     */
    if (temp[str[j]] >= i && str[j] in temp) {
      i = temp[str[j]] + 1;
    }
    temp[str[j]] = j; // 以 str[j] 结尾的子字符串的结尾的索引为 j
    result = Math.max(result, j - i + 1); // 计算最大值
  }
  return result;
}
